-- Table Definition for doppio_bills (Sales/Invoices) in Supabase
CREATE TABLE IF NOT EXISTS public.doppio_bills (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL,
    "orderId" text UNIQUE NOT NULL,
    "customerName" text DEFAULT 'Walk-in Guest'::text,
    "customerPhone" text,
    "dateTime" text,
    items text, -- JSON stringified items
    subtotal numeric DEFAULT 0,
    gst numeric DEFAULT 0,
    total numeric DEFAULT 0,
    "paymentMethod" text DEFAULT 'UPI'::text
);

-- Table Definition for doppio_draft_orders (Active Held Carts) in Supabase
CREATE TABLE IF NOT EXISTS public.doppio_draft_orders (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    "draftId" text UNIQUE NOT NULL,
    "draftName" text NOT NULL,
    "customerName" text,
    "customerPhone" text,
    "paymentMethod" text DEFAULT 'UPI'::text,
    items text NOT NULL, -- JSON stringified items
    subtotal numeric DEFAULT 0,
    gst numeric DEFAULT 0,
    total numeric DEFAULT 0,
    "createdAt" timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Enable Row Level Security (Mandatory for Commercial POS Data Privacy)
ALTER TABLE public.doppio_bills ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.doppio_draft_orders ENABLE ROW LEVEL SECURITY;

-- Create secure policies (Only authenticated staff accounts can read/edit invoices and active held carts)
DROP POLICY IF EXISTS "Allow authenticated staff to read bills" ON public.doppio_bills;
CREATE POLICY "Allow authenticated staff to read bills" ON public.doppio_bills FOR SELECT TO authenticated USING (true);

DROP POLICY IF EXISTS "Allow authenticated staff to insert bills" ON public.doppio_bills;
CREATE POLICY "Allow authenticated staff to insert bills" ON public.doppio_bills FOR INSERT TO authenticated WITH CHECK (true);

DROP POLICY IF EXISTS "Allow authenticated staff to delete bills" ON public.doppio_bills;
CREATE POLICY "Allow authenticated staff to delete bills" ON public.doppio_bills FOR DELETE TO authenticated USING (true);

DROP POLICY IF EXISTS "Allow authenticated staff to manage drafts" ON public.doppio_draft_orders;
CREATE POLICY "Allow authenticated staff to manage drafts" ON public.doppio_draft_orders FOR ALL TO authenticated USING (true);

-- Create indexes for super-fast lookups
CREATE INDEX IF NOT EXISTS idx_doppio_bills_order_id ON public.doppio_bills("orderId");
CREATE INDEX IF NOT EXISTS idx_doppio_bills_date ON public.doppio_bills("dateTime");
CREATE INDEX IF NOT EXISTS idx_doppio_drafts_id ON public.doppio_draft_orders("draftId");

-- Table Definition for doppio_shifts (Cashier Sessions) in Supabase
CREATE TABLE IF NOT EXISTS public.doppio_shifts (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL,
    "shiftId" text UNIQUE NOT NULL,
    "cashierName" text NOT NULL,
    "openedAt" timestamp with time zone NOT NULL,
    "closedAt" timestamp with time zone,
    "openingFloat" numeric DEFAULT 0 NOT NULL,
    "expectedCash" numeric DEFAULT 0,
    "actualCash" numeric DEFAULT 0,
    variance numeric DEFAULT 0,
    "totalSalesCash" numeric DEFAULT 0,
    "totalSalesUpi" numeric DEFAULT 0,
    "totalSalesCard" numeric DEFAULT 0,
    "totalPayouts" numeric DEFAULT 0,
    "totalSafeDrops" numeric DEFAULT 0,
    status text DEFAULT 'OPEN'::text NOT NULL,
    notes text
);

-- Table Definition for doppio_shift_events (Drawer Payouts & Drops) in Supabase
CREATE TABLE IF NOT EXISTS public.doppio_shift_events (
    id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL,
    "eventId" text UNIQUE NOT NULL,
    "shiftId" text NOT NULL,
    "eventType" text NOT NULL,
    amount numeric NOT NULL,
    reason text NOT NULL
);

-- Enable Row Level Security for Shifts and Adjustment Events
ALTER TABLE public.doppio_shifts ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.doppio_shift_events ENABLE ROW LEVEL SECURITY;

DROP POLICY IF EXISTS "Allow authenticated staff to manage shifts" ON public.doppio_shifts;
CREATE POLICY "Allow authenticated staff to manage shifts" ON public.doppio_shifts FOR ALL TO authenticated USING (true);

DROP POLICY IF EXISTS "Allow authenticated staff to manage shift events" ON public.doppio_shift_events;
CREATE POLICY "Allow authenticated staff to manage shift events" ON public.doppio_shift_events FOR ALL TO authenticated USING (true);

-- Create indexes for shifts session auditing
CREATE INDEX IF NOT EXISTS idx_doppio_shifts_id ON public.doppio_shifts("shiftId");
CREATE INDEX IF NOT EXISTS idx_doppio_shift_events_shift ON public.doppio_shift_events("shiftId");
